#1、查询出部门编号为30的所有员工编号和姓名：
select empno,ename from emp
where deptno=30;

#2、找出部门编号为10的所有经理和部门编号为30的所有销售员的详细资料：
select * from ruozedata.emp
where (deptno=10 and job=‘MANAGER’)
or
(deptno=30 and job=‘SALESMAN’);

#3、查出所有员工详细信息，用工资降序排序，如果工资相同使用入职日期升序排序：
select * from ruozedata.emp
group by sal desc,
hiredate desc;

#列出最低薪金大于1500的各种工作及从事此工作的员工人数：
use ruozedata;
select * from emp;

#我们看到各这种表示是要做group by语句的:
#注意：null值加上任意值都是null
#直观展示，作出修改：是null值字段，把它变作0；
select job,sal+comm,sal+ifnull(comm,0) from emp;

select job,count(job) from ruozedata.emp where (sal+ifnull(comm,0)) > 1500
group by job;

#最低薪金？
select job,count(job) from ruozedata.emp
group by job
having min(sal+ifnull(comm,0)) > 1500;

#5、列出在销售部工作的员工姓名，假设不知道销售部的部门编号：
#思路：先查出销售部门编号，再去员工表查找出来即可。
select ename from ruozedata.emp
where deptno=
(select deptno from dept where dname=‘SALES’);

#6、查询姓名以S开头、以S结尾、包含S字符、第二个字母为L
select ename from emp where ename like ‘S%’;
select ename from emp where ename like ‘%S’;
select ename from emp where ename like ‘%S%’;
select ename from emp where ename like ‘_L%’;

#7、查询每种工作的最高工资、最低工资、及人数
select emp.job,max(sal),min(sal),count(1) as 人数
from emp group by emp.job;

#8、列出薪金 高于公司平均薪金的所有员工号，员工姓名，所在部门名称，上级领导，工资，工资等级：
#在emp表中只能获取这些信息，emp表全量数据去匹配dept表，条件是deptno相等

select
ename,
deptno,
mgr,
sal+ifnull(comm,0) as sal
from ruozedata.emp as e
left join ruozedata.dept as d on e.deptno=d.deptno
where (sal+ifnull(comm,0)) > (select avg(sal+ifnull(comm,0)) from emp);

select e.empno as 工号,e.ename as 姓名,
d.dname as 部门,
m.ename as 上级领导,
(e.sal+ifnull(e.comm,0)) as 工资,
s.grade as 薪水等级
from emp e
left join dept d on e.deptno=d.deptno
left join emp m on e.mgr=m.empno
left join salgrade s on e.sal between s.losal and s.hisal
where (e.sal+ifnull(e.comm,0)) > (select avg(sal+ifnull(comm,0)) from emp);

9.列出薪金 高于 在部门30工作的 所有/任何一个员工的薪金的员工姓名和薪金、部门名称。
select e.ename, e.sal, d.dname
from emp e, dept d
where e.deptno=d.deptno and sal > all(select sal from emp where deptno=30)

any:
select e.ename, e.sal, d.dname
from emp e, dept d
where e.deptno=d.deptno and sal > any(select sal from emp where deptno=30)
